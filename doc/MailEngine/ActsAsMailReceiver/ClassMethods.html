def acts_as_mail_receiver(options)
  [:payload_columns, :groups].each do |key|
    methods_array = Array.wrap(options[key]) || []
    self.send("#{key}=", methods_array)
  end
  check_method_existance! unless options[:skip_method_existance_check]
end
